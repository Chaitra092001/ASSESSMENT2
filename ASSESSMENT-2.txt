1)EXEC SP_HELP LMS_BOOK_DETAILS ;

2)SELECT * FROM LMS_FINE_DETAILS;

3)SELECT MEMBER_NAME 
FROM LMS_MEMBERS
WHERE CITY='CHENNAI';

4)SELECT MEMBER_ID,MEMBER_NAME, CITY,MEMBERSHIP_STATUS
FROM LMS_MEMBERS
WHERE MEMBERSHIP_STATUS='PERMANENT';

5)SELECT M.MEMBER_ID,MEMBER_NAME
FROM LMS_MEMBERS M, LMS_BOOK_ISSUE B
WHERE M.MEMBER_ID=B.MEMBER_ID AND B.BOOK_CODE='BL000002';

6)SELECT BOOK_CODE, BOOK_TITLE, AUTHOR
FROM LMS_BOOK_DETAILS
WHERE AUTHOR LIKE 'P%';

7)SELECT COUNT(*) AS NO_OF_BOOKS
FROM LMS_BOOK_DETAILS
WHERE BOOK_TITLE LIKE 'JAVA%';

8)SELECT CATEGORY ,  COUNT(*) AS NO_OF_BOOKS
FROM LMS_BOOK_DETAILS
GROUP BY CATEGORY;

9)SELECT COUNT(*) AS NO_OF_BOOKS
FROM LMS_BOOK_DETAILS
WHERE PUBLICATION='PRENTICE HALL';

10)SELECT B.BOOK_CODE, BOOK_TITLE 
FROM LMS_BOOK_DETAILS B, LMS_BOOK_ISSUE I
WHERE B.BOOK_CODE=I.BOOK_CODE AND DATE_ISSUE='2012-04-01';

11)SELECT MEMBER_ID, MEMBER_NAME, DATE_REGISTER, DATE_EXPIRE
FROM LMS_MEMBERS
WHERE DATE_EXPIRE  < '2013-04-01';

12)SELECT MEMBER_ID, MEMBER_NAME, DATE_REGISTER, MEMBERSHIP_STATUS
FROM LMS_MEMBERS
WHERE MEMBERSHIP_STATUS='TEMPORARY' AND DATE_REGISTER<'2012-03-01';

13)SELECT MEMBER_ID, MEMBER_NAME
FROM LMS_MEMBERS
WHERE CITY='CHENNAI' OR CITY='DELHI';

14)SELECT BOOK_TITLE+ ' IS_WRITTEN_BY '+AUTHOR AS 'TITLE AND AUTHOR'
FROM LMS_BOOK_DETAILS

15)SELECT AVG(PRICE) AS AVERAGEPRICE
FROM LMS_BOOK_DETAILS
WHERE CATEGORY='JAVA';

16)SELECT SUPPLIER_ID, SUPPLIER_NAME, EMAIL 
FROM LMS_SUPPLIERS_DETAILS
WHERE EMAIL LIKE '%@gmail.com';

17) SELECT SUPPLIER_ID, SUPPLIER_NAME, COALESCE(CONTACT, EMAIL, ADDRESS) AS CONTACTDETAILS
 FROM LMS_SUPPLIERS_DETAILS;

18)SELECT DISTINCT M.MEMBER_ID, MEMBER_NAME, B.BOOK_CODE, BOOK_TITLE
FROM LMS_BOOK_DETAILS B, LMS_MEMBERS M, LMS_BOOK_ISSUE I
WHERE M.MEMBER_ID=I.MEMBER_ID  AND B.BOOK_CODE=I.BOOK_CODE;

19)SELECT M.MEMBER_ID, MEMBER_NAME, F.FINE_RANGE, FINE_AMOUNT
 FROM LMS_BOOK_ISSUE I, LMS_MEMBERS M, LMS_FINE_DETAILS F
 WHERE M.MEMBER_ID=I.MEMBER_ID AND I.FINE_RANGE=F.FINE_RANGE AND FINE_AMOUNT<100;

20)SELECT BOOK_CODE, BOOK_TITLE, RACK_NUM
FROM LMS_BOOK_DETAILS
WHERE RACK_NUM='A1'
ORDER BY BOOK_TITLE ASC;

21)SELECT M.MEMBER_ID,MEMBER_NAME,DATE_REGISTER 
 FROM LMS_MEMBERS M LEFT JOIN LMS_BOOK_ISSUE B ON M.MEMBER_ID = B.MEMBER_ID WHERE B.MEMBER_ID IS NULL;

22)  SELECT M.MEMBER_ID, MEMBER_NAME
 FROM LMS_MEMBERS M JOIN LMS_BOOK_ISSUE B ON M.MEMBER_ID=B.MEMBER_ID 
 WHERE DATE_RETURNED<=DATE_RETURN AND DATE_RETURNED LIKE '2012%';

23)SELECT RACK_NUM, COUNT(*) AS NOOFBOOKS
 FROM LMS_BOOK_DETAILS
 GROUP BY RACK_NUM
 ORDER BY RACK_NUM ASC

24) SELECT I.BOOK_ISSUE_NO, MEMBER_NAME, DATE_REGISTER , DATE_EXPIRE, BOOK_TITLE, CATEGORY,AUTHOR, PRICE, DATE_ISSUE, DATE_RETURN, DATE_RETURNED,'ISSUED' AS 'ISSUE_STATUS', FINE_AMOUNT
 FROM LMS_MEMBERS M  JOIN LMS_BOOK_ISSUE I ON M.MEMBER_ID=I.MEMBER_ID 
 JOIN LMS_BOOK_DETAILS B ON B.BOOK_CODE=I.BOOK_CODE 
 JOIN LMS_FINE_DETAILS  F ON F.FINE_RANGE=I.FINE_RANGE;

25)SELECT BI.MEMBER_ID,MEMBER_NAME,3-COUNT(BI.MEMBER_ID) AS REMAININGBOOKS
FROM LMS_BOOK_ISSUE BI,LMS_MEMBERS M WHERE M.MEMBER_ID=BI.MEMBER_ID
GROUP BY BI.MEMBER_ID,M.MEMBER_NAME;


26)SELECT TOP 3 * FROM LMS_SUPPLIERS_DETAILS;

27)SELECT * FROM LMS_SUPPLIERS_DETAILS 
WHERE SUPPLIER_ID NOT IN  ('S01','S02','S03');

28)EXEC SP_RENAME 'LMS_MEMBERS' , 'LMS_MEMBERS_DETAILS';

29)TRUNCATE TABLE LMS_FINE_DETAILS

30) SELECT ALL BOOK_TITLE AS NAME FROM LMS_BOOK_DETAILS;
